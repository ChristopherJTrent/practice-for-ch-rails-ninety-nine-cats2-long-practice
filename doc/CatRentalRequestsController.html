def create
  @rental_request = CatRentalRequest.new(cat_rental_request_params)
  if @rental_request.save
    redirect_to cat_url(@rental_request.cat)
  else
    flash.now[:errors] = @rental_request.errors.full_messages
    render :new
  end
end
